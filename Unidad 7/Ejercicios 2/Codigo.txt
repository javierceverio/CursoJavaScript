        var estadoClima = (temperatura) => {
            if (temperatura > 25) {
                return "Cálido";
            }
            else if (temperatura < 25 && temperatura > 10) {
                return "Templado";
            }
            else {
                return "Frío";
            }
        }
        var temperatura = parseFloat(prompt("Ingrese la temperatura de su ciudad:"));
        console.log("La temperatura de hoy es " + temperatura + " grados, entonces el clima está: " + estadoClima(temperatura));

        var nombre;
        var edad;
        var sueldo;
        var persona = new Array();
        var personas = new Array();
        var cargarSueldos = () => {
            var cantidad = parseInt(prompt("Ingrese la cantidad de Empleados que va a cargar:"));
            for (i = 0; i < cantidad; i++) {
                nombre = prompt("Ingrese el Nombre del Empleado " + (i + 1) + ":");
                edad = parseInt(prompt("Ingrese la edad del Empleado " + (i + 1) + ":"));
                sueldo = parseFloat(prompt("Ingrese el Sueldo del Empleado " + (i + 1) + ":"));
                persona.push(nombre);
                persona.push(edad);
                persona.push(sueldo);
                personas.push(persona);
            }
        }
        var mostrarDatosEmpleado = (indice) => {
            console.log("Nombre: " + personas[indice][indice * 3] + ", Edad: " + personas[indice][(indice * 3) + 1] + ", Sueldo: $ " + personas[indice][(indice * 3) + 2]);
        }
        function mostrarTodosLosEmpleados() {
            for (i = 0; i < personas.length; i++){
                console.log("--------------------------------------------");
                console.log("Empleado " + (i + 1) + ": ");
                mostrarDatosEmpleado(i);
            }
        }
        cargarSueldos();
        mostrarDatosEmpleado(1);
        mostrarTodosLosEmpleados();
